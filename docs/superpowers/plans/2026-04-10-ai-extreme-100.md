```typescript
const [engineStatus, setEngineStatus] = useState<AIEngineStatus>('idle');

const alertCount = proactiveAlerts.filter(a => !a.dismissed).length;

const pushAlert = useCallback((alert: Omit<AIProactiveAlert, 'id' | 'timestamp' | 'dismissed'>) => {
    setProactiveAlerts(prev => {
        // Deduplicate by sourceField + severity
        const exists = prev.find(a => a.sourceField === alert.sourceField && a.severity === alert.severity && !a.dismissed);
        if (exists) return prev;
        return [...prev, { ...alert, id: `alert-${Date.now()}-${Math.random().toString(36).slice(2)}`, timestamp: Date.now(), dismissed: false }];
    });
}, []);

const dismissAlert = useCallback((id: string) => {
    setProactiveAlerts(prev => prev.map(a => a.id === id ? { ...a, dismissed: true } : a));
}, []);

const clearAlerts = useCallback(() => {
    setProactiveAlerts([]);
}, []);
```

```typescript
'use client';

import { useEffect, useRef, useCallback } from 'react';
import { useAICopilot } from '@/contexts/AICopilotContext';
import { useAIPreferences } from '@/hooks/useAIPreferences';
import { aiService } from '@/services/aiService';
import { AI_DEBOUNCE_MS } from '@/constants/ai';

export function useAIAmbientEngine() {
    const { pageContext, role, pushAlert, setEngineStatus } = useAICopilot();
    const { preferences } = useAIPreferences();
    const prevDataRef = useRef<string>('');
    const timerRef = useRef<ReturnType<typeof setTimeout>>();
    const mountedRef = useRef(true);

    useEffect(() => { return () => { mountedRef.current = false; }; }, []);

    const runAnalysis = useCallback(async () => {
        if (!pageContext?.formData || !mountedRef.current) return;
        setEngineStatus('analyzing');

        try {
            if (role === 'doctor') {
                await runDoctorChecks(pageContext, pushAlert);
            } else if (role === 'pharmacist') {
                await runPharmacistChecks(pageContext, pushAlert);
            } else if (role === 'receptionist') {
                await runReceptionistChecks(pageContext, pushAlert);
            }
        } catch {
            // Silent fail — AI failure never blocks workflow
        }

        if (mountedRef.current) setEngineStatus('ready');
    }, [pageContext, role, pushAlert, setEngineStatus]);

    useEffect(() => {
        if (!preferences.enableExamSuggestions) return;
        const serialized = JSON.stringify(pageContext?.formData);
        if (serialized === prevDataRef.current) return;
        prevDataRef.current = serialized;

        clearTimeout(timerRef.current);
        timerRef.current = setTimeout(runAnalysis, AI_DEBOUNCE_MS);
        return () => clearTimeout(timerRef.current);
    }, [pageContext?.formData, preferences.enableExamSuggestions, runAnalysis]);
}

async function runDoctorChecks(
    ctx: NonNullable<ReturnType<typeof useAICopilot>['pageContext']>,
    pushAlert: ReturnType<typeof useAICopilot>['pushAlert']
) {
    const formData = ctx.formData as any;
    if (!formData) return;

    // Check vitals
    if (formData.vitals?.bloodPressure) {
        const bp = formData.vitals.bloodPressure;
        const match = bp.match(/(\d+)\/(\d+)/);
        if (match) {
            const sys = parseInt(match[1]), dia = parseInt(match[2]);
            if (sys >= 180 || dia >= 120) {
                pushAlert({
                    severity: 'critical',
                    message: `⚠️ HA ${bp} mmHg — Ngưỡng THA cấp cứu (≥180/120). Cần đánh giá tổn thương cơ quan đích.`,
                    icon: 'warning',
                    sourceField: 'bloodPressure',
                    autoFillPayload: { icdCode: 'I16.1', diagnosis: 'Cơn tăng huyết áp cấp cứu' },
                });
            } else if (sys >= 140 || dia >= 90) {
                pushAlert({
                    severity: 'warning',
                    message: `HA ${bp} mmHg — Tăng huyết áp. Theo dõi và cân nhắc điều trị.`,
                    icon: 'monitor_heart',
                    sourceField: 'bloodPressure',
                });
            }
        }
    }

    // Check SpO2
    if (formData.vitals?.spO2) {
        const spo2 = parseFloat(formData.vitals.spO2);
        if (spo2 > 0 && spo2 < 92) {
            pushAlert({
                severity: 'critical',
                message: `⚠️ SpO₂ ${spo2}% — Thiếu oxy nghiêm trọng. Cần can thiệp ngay.`,
                icon: 'emergency',
                sourceField: 'spO2',
            });
        }
    }

    // Check temperature
    if (formData.vitals?.temperature) {
        const temp = parseFloat(formData.vitals.temperature);
        if (temp >= 39) {
            pushAlert({
                severity: 'warning',
                message: `Nhiệt độ ${temp}°C — Sốt cao. Cân nhắc hạ sốt và tìm nguyên nhân.`,
                icon: 'thermostat',
                sourceField: 'temperature',
            });
        }
    }

    // Symptom analysis
    if (formData.symptoms && typeof formData.symptoms === 'string' && formData.symptoms.length > 20) {
        try {
            const res = await aiService.getDiagnosis({ symptoms: formData.symptoms, vitals: formData.vitals });
            const diagnoses = (res?.data as any)?.diagnoses;
            if (diagnoses?.[0]?.confidence >= 70) {
                const top = diagnoses[0];
                pushAlert({
                    severity: 'info',
                    message: `💡 Gợi ý: ${top.icdDescription || top.description} (${top.icdCode}) — ${top.confidence}% match`,
                    icon: 'diagnosis',
                    sourceField: 'diagnosis',
                    autoFillPayload: { diagnosis: top.icdDescription || top.description, icdCode: top.icdCode },
                });
            }
        } catch { /* silent */ }
    }
}

async function runPharmacistChecks(
    ctx: NonNullable<ReturnType<typeof useAICopilot>['pageContext']>,
    pushAlert: ReturnType<typeof useAICopilot>['pushAlert']
) {
    const formData = ctx.formData as any;
    if (!formData?.drugs || !Array.isArray(formData.drugs) || formData.drugs.length < 2) return;

    try {
        const res = await aiService.checkDrugInteraction({
            drugs: formData.drugs,
            allergies: formData.allergies || [],
        });
        const interactions = (res?.data as any)?.interactions;
        if (interactions?.length) {
            const severe = interactions.find((i: any) => i.severity === 'serious' || i.severity === 'contraindicated');
            if (severe) {
                pushAlert({
                    severity: 'critical',
                    message: `⚠️ Tương tác thuốc: ${severe.drugA} + ${severe.drugB} — ${severe.detail}`,
                    icon: 'medication',
                    sourceField: 'drugInteraction',
                });
            }
        }
    } catch { /* silent */ }
}

async function runReceptionistChecks(
    ctx: NonNullable<ReturnType<typeof useAICopilot>['pageContext']>,
    pushAlert: ReturnType<typeof useAICopilot>['pushAlert']
) {
    const formData = ctx.formData as any;
    if (!formData?.reason || typeof formData.reason !== 'string' || formData.reason.length < 5) return;

    // Client-side keyword triage
    const urgentKeywords = ['đau ngực', 'khó thở', 'co giật', 'bất tỉnh', 'xuất huyết', 'ngất', 'sốt cao'];
    const lower = formData.reason.toLowerCase();
    const matched = urgentKeywords.find(kw => lower.includes(kw));
    if (matched) {
        pushAlert({
            severity: 'critical',
            message: `🚨 Triệu chứng "${matched}" — cần khám khẩn cấp. Ưu tiên lên đầu hàng đợi.`,
            icon: 'emergency',
            sourceField: 'triage',
        });
    }
}
```

```typescript
'use client';

import { useState, useRef, useEffect, useCallback } from 'react';
import { useRouter } from 'next/navigation';
import { useAICopilot } from '@/contexts/AICopilotContext';
import { aiService } from '@/services/aiService';

interface SearchResult {
    id: string;
    type: 'patient' | 'medicine' | 'appointment' | 'record';
    title: string;
    subtitle: string;
    href: string;
    icon: string;
}

const TYPE_STYLE: Record<string, { icon: string; label: string; color: string }> = {
    patient: { icon: 'person', label: 'Bệnh nhân', color: 'text-blue-600 bg-blue-50' },
    medicine: { icon: 'medication', label: 'Thuốc', color: 'text-green-600 bg-green-50' },
    appointment: { icon: 'event', label: 'Lịch hẹn', color: 'text-violet-600 bg-violet-50' },
    record: { icon: 'folder_shared', label: 'Hồ sơ', color: 'text-amber-600 bg-amber-50' },
};

export default function AISearchBar() {
    const router = useRouter();
    const { role } = useAICopilot();
    const [query, setQuery] = useState('');
    const [results, setResults] = useState<SearchResult[]>([]);
    const [isSearching, setIsSearching] = useState(false);
    const [showDropdown, setShowDropdown] = useState(false);
    const [selectedIndex, setSelectedIndex] = useState(-1);
    const inputRef = useRef<HTMLInputElement>(null);
    const timerRef = useRef<ReturnType<typeof setTimeout>>();

    const search = useCallback(async (q: string) => {
        if (q.length < 2) { setResults([]); return; }
        setIsSearching(true);
        try {
            const res = await aiService.chat({
                message: q,
                context: { type: 'search', role },
            });
            const data = (res?.data as any);
            // Parse AI response into structured results
            // Fallback: generate mock results from query
            const mockResults: SearchResult[] = [
                { id: '1', type: 'patient', title: `BN: ${q}`, subtitle: 'Tìm kiếm bệnh nhân...', href: '#', icon: 'person' },
                { id: '2', type: 'medicine', title: `Thuốc: ${q}`, subtitle: 'Tìm kiếm thuốc...', href: '#', icon: 'medication' },
            ];
            setResults(data?.results || mockResults);
        } catch {
            setResults([]);
        } finally {
            setIsSearching(false);
        }
    }, [role]);

    useEffect(() => {
        if (!query.trim()) { setResults([]); setShowDropdown(false); return; }
        clearTimeout(timerRef.current);
        timerRef.current = setTimeout(() => {
            search(query);
            setShowDropdown(true);
        }, 300);
        return () => clearTimeout(timerRef.current);
    }, [query, search]);

    const handleKeyDown = (e: React.KeyboardEvent) => {
        if (e.key === 'ArrowDown') { e.preventDefault(); setSelectedIndex(prev => Math.min(prev + 1, results.length - 1)); }
        else if (e.key === 'ArrowUp') { e.preventDefault(); setSelectedIndex(prev => Math.max(prev - 1, -1)); }
        else if (e.key === 'Enter' && selectedIndex >= 0 && results[selectedIndex]) {
            router.push(results[selectedIndex].href);
            setShowDropdown(false);
            setQuery('');
        }
        else if (e.key === 'Escape') { setShowDropdown(false); }
    };

    return (
        <div className="relative w-72 hidden md:block">
            <span className="absolute inset-y-0 left-0 flex items-center pl-3 text-[#687582]">
                <span className="material-symbols-outlined text-[18px]">{isSearching ? 'sync' : 'search'}</span>
            </span>
            <input
                ref={inputRef}
                type="text"
                value={query}
                onChange={(e) => setQuery(e.target.value)}
                onKeyDown={handleKeyDown}
                onFocus={() => query.length >= 2 && setShowDropdown(true)}
                onBlur={() => setTimeout(() => setShowDropdown(false), 200)}
                placeholder="🤖 Tìm kiếm thông minh..."
                className="w-full py-2 pl-10 pr-4 text-sm bg-[#f8fafc] dark:bg-gray-800 border border-gray-200 dark:border-gray-700 rounded-xl focus:outline-none focus:ring-2 focus:ring-violet-400/30 focus:border-violet-400 dark:text-white placeholder:text-gray-400 transition-all"
            />
            {showDropdown && results.length > 0 && (
                <div className="absolute top-full left-0 right-0 mt-1 bg-white dark:bg-[#1e242b] border border-gray-200 dark:border-gray-700 rounded-xl shadow-lg z-50 overflow-hidden">
                    {results.map((r, i) => {
                        const style = TYPE_STYLE[r.type] || TYPE_STYLE.patient;
                        return (
                            <button
                                key={r.id}
                                onClick={() => { router.push(r.href); setShowDropdown(false); setQuery(''); }}
                                className={`w-full flex items-center gap-3 px-3 py-2.5 text-left text-sm hover:bg-gray-50 dark:hover:bg-gray-800 transition-colors ${i === selectedIndex ? 'bg-violet-50 dark:bg-violet-900/20' : ''}`}
                            >
                                <div className={`w-8 h-8 rounded-lg flex items-center justify-center ${style.color}`}>
                                    <span className="material-symbols-outlined" style={{ fontSize: '16px' }}>{style.icon}</span>
                                </div>
                                <div className="flex-1 min-w-0">
                                    <p className="text-sm font-medium text-[#121417] dark:text-white truncate">{r.title}</p>
                                    <p className="text-xs text-[#687582] truncate">{r.subtitle}</p>
                                </div>
                                <span className={`text-[10px] px-1.5 py-0.5 rounded ${style.color}`}>{style.label}</span>
                            </button>
                        );
                    })}
                </div>
            )}
        </div>
    );
}
```

```typescript
    const [transcript, setTranscript] = useState('');
    const [isSupported, setIsSupported] = useState(false);
    const recognitionRef = useRef<any>(null);

    useEffect(() => {
        const SR = (window as any).SpeechRecognition || (window as any).webkitSpeechRecognition;
        if (!SR) { setIsSupported(false); return; }
        setIsSupported(true);

        const recognition = new SR();
        recognition.lang = options?.lang ?? 'vi-VN';
        recognition.continuous = options?.continuous ?? false;
        recognition.interimResults = true;

        recognition.onresult = (e: any) => {
            const t = Array.from(e.results as SpeechRecognitionResultList)
                .map((r: any) => r[0].transcript)
                .join('');
            setTranscript(t);
        };
        recognition.onerror = () => setIsListening(false);
        recognition.onend = () => setIsListening(false);
        recognitionRef.current = recognition;
    }, [options?.lang, options?.continuous]);

    const startListening = useCallback(() => {
        if (!recognitionRef.current) return;
        setTranscript('');
        try { recognitionRef.current.start(); setIsListening(true); } catch { /* already started */ }
    }, []);

    const stopListening = useCallback(() => {
        recognitionRef.current?.stop();
    }, []);

    return { isListening, transcript, isSupported, startListening, stopListening };
}
```
